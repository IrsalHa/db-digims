create PROCEDURE COUNTSERIALSTATUS_SP (
    p_totaldata OUT NUMBER,
    p_outstanding OUT NUMBER,
    p_injected OUT NUMBER,
    p_outproduksi OUT NUMBER
)
IS
BEGIN
    SELECT
        COUNT(*) AS totaldata,
        COUNT(CASE WHEN status = 'S01' THEN 1 END) AS outstanding,
        COUNT(CASE WHEN status = 'S11' THEN 1 END) AS injected,
        COUNT(CASE WHEN status = 'S13' THEN 1 END) AS outproduksi
    INTO 
        p_totaldata,
        p_outstanding,
        p_injected,
        p_outproduksi
    FROM Ttokenitem;
END COUNTSERIALSTATUS_SP;
/

create PROCEDURE            "GET_PRODUCTTYPE_BY_FILTER" (
    p_mbranchpk IN NUMBER,             -- Parameter input: ID Cabang
    p_productgroupcode IN VARCHAR,  -- Parameter input: Kode Grup Produk
    p_producttype IN VARCHAR,        -- Parameter input: Product Type (LIKE)
    p_cursor OUT SYS_REFCURSOR         -- Output: Cursor untuk hasil query
) AS
BEGIN
    OPEN p_cursor FOR
    SELECT 
        MPRODUCTTYPEPK,
        TRIM(productgroupcode) AS productgroupcode,
        TRIM(productgroupname) AS productgroupname,
        TRIM(producttype) AS producttype
    FROM mproducttype p
    JOIN mproductowner o ON p.mproductownerfk = o.mproductownerpk
    JOIN mbranch b ON b.mbranchpk = o.mbranchfk
    WHERE b.mbranchpk = p_mbranchpk
    AND LOWER(TRIM(p.productgroupcode)) = p_productgroupcode
    AND LOWER(TRIM(p.producttype)) LIKE LOWER(TRIM(p_producttype));
END;
/

create PROCEDURE            "LIST_PRODUCT_STOCK" (
    param IN VARCHAR2,        
    result OUT SYS_REFCURSOR
) AS
BEGIN
    OPEN result  FOR
        'SELECT MBRANCHPAGUPK, BRANCHNAME, PRODUCTNAME, QUOTA, COALESCE(PAGU, 0) PAGU, COALESCE(STOCK, 0) AS STOCK, 
        ((COALESCE(STOCK, 0) / COALESCE(PAGU, 0)) * 100) AS PERCENT FROM
        (SELECT MBRANCHPAGUPK, MBRANCHFK, MPRODUCTFK, BRANCHNAME, PRODUCTNAME, COALESCE(QUOTA, 0) AS QUOTA, COALESCE(PAGU, 0) AS PAGU
        FROM MBRANCHPAGU JOIN MPRODUCT ON MPRODUCTPK = MPRODUCTFK JOIN MBRANCH ON MBRANCHFK = MBRANCHPK
        WHERE ' || param || ') A LEFT JOIN
        (SELECT MBRANCHFK, MPRODUCTFK,  COALESCE(STOCKCABANG, 0) AS STOCK FROM TBRANCHSTOCK WHERE OUTLET = ''00'') B
        ON A.MBRANCHFK = B.MBRANCHFK AND A.MPRODUCTFK = B.MPRODUCTFK';
END;
/

create PROCEDURE REPORT_MONTHLY_REGULAR (
    FILTERS IN VARCHAR2,
    RESULT_CURSOR OUT SYS_REFCURSOR
)
IS
    v_sql CLOB;
BEGIN
    v_sql := '
        SELECT 
            a.mproductfk AS id,
            a.productorg,
            a.producttype,
            a.productcode,
            a.productname,
            COALESCE(total, 0) AS total,
            COALESCE(perso, 0) AS perso,
            COALESCE(pending, 0) AS pending,
            COALESCE(delivery, 0) AS delivery
        FROM (
            SELECT 
                tembossdata.mproductfk,
                productorg,
                producttype,
                tembossdata.productcode,
                productname,
                COUNT(tembossdatapk) AS total
            FROM tembossdata
            JOIN mproduct ON tembossdata.mproductfk = mproductpk
            JOIN mbranch ON tembossdata.mbranchfk = mbranchpk
            JOIN tembossbranch ON tembossbranchfk = tembossbranchpk
            JOIN mproducttype ON mproducttypefk = mproducttypepk
            WHERE ' || FILTERS || '
              AND tembossdata.mproductfk IS NOT NULL
            GROUP BY tembossdata.mproductfk, productorg, producttype, tembossdata.productcode, productname
        ) a
        LEFT JOIN (
            SELECT 
                tembossdata.mproductfk,
                productorg,
                producttype,
                tembossdata.productcode,
                productname,
                COUNT(tembossdatapk) AS pending
            FROM tembossdata
            JOIN mproduct ON tembossdata.mproductfk = mproductpk
            JOIN mbranch ON tembossdata.mbranchfk = mbranchpk
            JOIN tembossbranch ON tembossbranchfk = tembossbranchpk
            JOIN mproducttype ON mproducttypefk = mproducttypepk
            WHERE ' || FILTERS || '
              AND tembossbranch.status = ''P01''
              AND tembossdata.mproductfk IS NOT NULL
            GROUP BY tembossdata.mproductfk, productorg, producttype, tembossdata.productcode, productname
        ) b ON a.mproductfk = b.mproductfk
        LEFT JOIN (
            SELECT 
                tembossdata.mproductfk,
                productorg,
                producttype,
                tembossdata.productcode,
                productname,
                COUNT(tembossdatapk) AS perso
            FROM tembossdata
            LEFT JOIN tderivatifdata ON tembossdatafk = tembossdatapk
            JOIN mproduct ON tembossdata.mproductfk = mproductpk
            JOIN mbranch ON tembossdata.mbranchfk = mbranchpk
            JOIN tembossbranch ON tembossbranchfk = tembossbranchpk
            JOIN tperso ON tembossdata.tembossproductfk = tperso.tembossproductfk
            JOIN mproducttype ON mproducttypefk = mproducttypepk
            WHERE ' || FILTERS || '
              AND persofinishtime IS NOT NULL
              AND tembossdata.mproductfk IS NOT NULL
            GROUP BY tembossdata.mproductfk, productorg, producttype, tembossdata.productcode, productname
        ) c ON a.mproductfk = c.mproductfk
        LEFT JOIN (
            SELECT 
                tembossdata.mproductfk,
                productorg,
                producttype,
                tembossdata.productcode,
                productname,
                COUNT(tembossdatapk) AS delivery
            FROM tembossdata
            JOIN mproduct ON tembossdata.mproductfk = mproductpk
            JOIN mbranch ON tembossdata.mbranchfk = mbranchpk
            JOIN tembossbranch ON tembossbranchfk = tembossbranchpk
            JOIN mproducttype ON mproducttypefk = mproducttypepk
            WHERE ' || FILTERS || '
              AND dlvfinishtime IS NOT NULL
              AND tembossdata.mproductfk IS NOT NULL
            GROUP BY tembossdata.mproductfk, productorg, producttype, tembossdata.productcode, productname
        ) d ON a.mproductfk = d.mproductfk
        ORDER BY a.productorg, a.producttype, a.productcode, a.productname';

    OPEN RESULT_CURSOR FOR v_sql;
END;
/
