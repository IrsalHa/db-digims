create or replace PROCEDURE COUNTSERIALSTATUS_SP (
    p_totaldata OUT NUMBER,
    p_outstanding OUT NUMBER,
    p_injected OUT NUMBER,
    p_outproduksi OUT NUMBER
)
IS
BEGIN
    SELECT
        COUNT(*) AS totaldata,
        COUNT(CASE WHEN status = 'S01' THEN 1 END) AS outstanding,
        COUNT(CASE WHEN status = 'S11' THEN 1 END) AS injected,
        COUNT(CASE WHEN status = 'S13' THEN 1 END) AS outproduksi
    INTO
        p_totaldata,
        p_outstanding,
        p_injected,
        p_outproduksi
    FROM Ttokenitem;
END COUNTSERIALSTATUS_SP;

create or replace PROCEDURE            "LIST_PRODUCT_STOCK" (
    param IN char,
    result OUT SYS_REFCURSOR
) AS
BEGIN
    OPEN result  FOR
        'SELECT MBRANCHPAGUPK, BRANCHNAME, PRODUCTNAME, QUOTA, COALESCE(PAGU, 0) PAGU, COALESCE(STOCK, 0) AS STOCK,
        ((COALESCE(STOCK, 0) / COALESCE(PAGU, 0)) * 100) AS PERCENT FROM
        (SELECT MBRANCHPAGUPK, MBRANCHFK, MPRODUCTFK, BRANCHNAME, PRODUCTNAME, COALESCE(QUOTA, 0) AS QUOTA, COALESCE(PAGU, 0) AS PAGU
        FROM MBRANCHPAGU JOIN MPRODUCT ON MPRODUCTPK = MPRODUCTFK JOIN MBRANCH ON MBRANCHFK = MBRANCHPK
        WHERE ' || param || ') A LEFT JOIN
        (SELECT MBRANCHFK, MPRODUCTFK,  COALESCE(STOCKCABANG, 0) AS STOCK FROM TBRANCHSTOCK WHERE OUTLET = ''00'') B
        ON A.MBRANCHFK = B.MBRANCHFK AND A.MPRODUCTFK = B.MPRODUCTFK';
END;
/

create or replace PROCEDURE            "GET_PRODUCTTYPE_BY_FILTER" (
    p_mbranchpk IN NUMBER,             -- Parameter input: ID Cabang
    p_productgroupcode IN VARCHAR,  -- Parameter input: Kode Grup Produk
    p_producttype IN VARCHAR,        -- Parameter input: Product Type (LIKE)
    p_cursor OUT SYS_REFCURSOR         -- Output: Cursor untuk hasil query
) AS
BEGIN
    OPEN p_cursor FOR
    SELECT
        MPRODUCTTYPEPK,
        TRIM(productgroupcode) AS productgroupcode,
        TRIM(productgroupname) AS productgroupname,
        TRIM(producttype) AS producttype
    FROM mproducttype p
    JOIN mproductowner o ON p.mproductownerfk = o.mproductownerpk
    JOIN mbranch b ON b.mbranchpk = o.mbranchfk
    WHERE b.mbranchpk = p_mbranchpk
    AND LOWER(TRIM(p.productgroupcode)) = p_productgroupcode
    AND LOWER(TRIM(p.producttype)) LIKE LOWER(TRIM(p_producttype));
END;
/

create or replace PROCEDURE            "LIST_SUM_REGULAR_ORDER" (
    result OUT SYS_REFCURSOR
) AS
BEGIN
    OPEN result  FOR
        'select extract(year from orderdate) as year, extract(month from orderdate) as month,
		sum(orderos) as total, org as productorg from tembossproduct where orderos > 0 and isneeddoc = ''N''
		and tembossproductpk not in (select tembossproductfk from tembossbranch join tembossproduct on tembossproductfk = tembossproductpk
		where (mbranchfk is null or tembossbranch.mproductfk is null) and isneeddoc = ''N'')
		group by extract(year from orderdate), extract(month from orderdate), org order by year, month';
END;
/

create or replace PROCEDURE            "LIST_STATUS_PENDING_PERSO" (
    result OUT SYS_REFCURSOR
) AS
BEGIN
    OPEN result  FOR
        'select extract(year from persostarttime) as year, extract(month from persostarttime) as month, status, count(*) as count, sum(totaldata) as sum
        from tperso where status NOT IN ('' P19 '', '' P12 '', '' P14 '') group by extract(YEAR from persostarttime), extract(month from persostarttime), status order by year, month, status';
END;

create or replace PROCEDURE            "LIST_DATA_PERSO_PLAN" (
    param IN char,
    result OUT SYS_REFCURSOR
) AS
BEGIN
    OPEN result  FOR
        'select productcode, sum(totaldata) as totaldata, min(orderdate) as orderdate from tperso
        join mproduct on mproductfk = mproductpk where status = ''P15'' and trunc(orderdate) = TO_DATE('''|| param ||''', ''YYYY-MM-DD'') group by productcode order by orderdate';
END;
